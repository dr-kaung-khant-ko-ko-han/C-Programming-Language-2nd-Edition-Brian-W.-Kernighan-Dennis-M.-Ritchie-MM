## ၅.၅ Character Pointer များနှင့် Function များ

String constant တစ်ခုကို

```
"I am a string"
```

ဟု ရေးထားပြီး ၎င်းသည် character များ၏ array တစ်ခုဖြစ်သည်။ Internal representation တွင် array သည် null character '\0' ဖြင့် အဆုံးသတ်ထားသောကြောင့် program များသည် အဆုံးကို ရှာတွေ့နိုင်သည်။ ထို့ကြောင့် storage တွင် အရှည်သည် double quote များကြားရှိ character များ၏ အရေအတွက်ထက် တစ်ခု ပို၍ရှိသည်။

String constant များ၏ အများဆုံး အသုံးများသောဖြစ်ရပ်မှာ function များသို့ argument များအဖြစ် ဖြစ်နိုင်သည်၊ ဥပမာအားဖြင့်

```
printf("hello, world\n");
```

ဤကဲ့သို့ character string တစ်ခု program တွင် ပေါ်လာသောအခါ ၎င်းသို့ ရယူအသုံးပြုခြင်းသည် character pointer တစ်ခုမှတဆင့်ဖြစ်သည်။ printf သည် character array ၏ အစသို့ pointer တစ်ခုကို လက်ခံရရှိသည်။ ဆိုလိုသည်မှာ string constant တစ်ခုကို ၎င်း၏ ပထမ element သို့ pointer တစ်ခုဖြင့် ရယူအသုံးပြုသည်။

String constant များသည် function argument များ ဖြစ်ရန် မလိုအပ်ပါ။ pmessage ကို

```
char *pmessage;
```

ဟု declare လုပ်ထားလျှင် statement

```
pmessage = "now is the time";
```

သည် pmessage သို့ character array သို့ pointer တစ်ခုကို သတ်မှတ်သည်။ ဤသည်မှာ string copy မဟုတ်ပါ၊ pointer များသာ ပါဝင်သည်။ C သည် character များ၏ string တစ်ခုလုံးကို unit တစ်ခုအဖြစ် လုပ်ဆောင်ရန် မည်သည့် operator မျှ မပံ့ပိုးပေးပါ။

ဤ definition များကြား အရေးကြီးသော ကွာခြားချက်တစ်ခုရှိသည်:

```
char amessage[] = "now is the time"; /* array တစ်ခု */

char *pmessage = "now is the time"; /* pointer တစ်ခု */ 
```

amessage သည် array တစ်ခုဖြစ်ပြီး ၎င်းကို initialize လုပ်သော character များ၏ sequence နှင့် '\0' ကို သိမ်းဆည်းရန် အလုံအလောက်သာ ကြီးသည်။ Array အတွင်းရှိ တစ်ခုချင်းစီသော character များကို ပြောင်းလဲနိုင်သော်လည်း amessage သည် အမြဲတူညီသော storage ကို ရည်ညွှန်းနေမည်ဖြစ်သည်။ အခြားတစ်ဖက်တွင် pmessage သည် pointer တစ်ခုဖြစ်ပြီး string constant တစ်ခုသို့ point လုပ်ရန် initialize လုပ်ထားသည်။ Pointer ကို နောက်ပိုင်းတွင် အခြားနေရာသို့ point လုပ်ရန် ပြင်ဆင်နိုင်သော်လည်း string အကြောင်းအရာများကို ပြင်ဆင်ရန် ကြိုးစားလျှင် ရလဒ်သည် သတ်မှတ်မထားသောအရာဖြစ်သည်။

Standard library မှ ပြင်ဆင်ထားသော အသုံးဝင်သော function နှစ်ခု၏ ဗားရှင်းများကို လေ့လာခြင်းဖြင့် pointer များနှင့် array များ၏ နောက်ထပ် ရှုထောင့်များကို ဖော်ပြပေးမည်။ ပထမ function မှာ strcpy(s,t) ဖြစ်ပြီး string t ကို string s သို့ ကူးယူသည်။ s=t ဟု ရိုးရိုးရှင်းရှင်း ဆိုနိုင်ရင် ကောင်းမည်ဖြစ်သော်လည်း ဤသည်မှာ pointer ကို ကူးယူရုံသာဖြစ်ပြီး character များကို ကူးယူခြင်းမဟုတ်ပါ။ Character များကို ကူးယူရန် loop တစ်ခု လိုအပ်သည်။ Array ဗားရှင်း ပထမဦးစွာ:

```
/* strcpy: t ကို s သို့ ကူးယူသည်; array subscript ဗားရှင်း */

void strcpy(char *s, char *t)

{
int i;
i = 0;

while ((s[i] = t[i]) != '\0')

	i++;
}
```

နှိုင်းယှဉ်ရန် ဤသည်မှာ pointer များပါသော strcpy ၏ ဗားရှင်းဖြစ်သည်:

```
/* strcpy: t ကို s သို့ ကူးယူသည်; pointer ဗားရှင်း */

void strcpy(char *s, char *t)

{
int i;
i = 0;

	while ((*s = *t) != '\0') {
		s++;
		t++;
	}
}
```

Argument များကို value ဖြင့် ပို့သောကြောင့် strcpy သည် parameter s နှင့် t တို့ကို သူကြိုက်သလို မည်သို့မဆို အသုံးပြုနိုင်သည်။ ဤနေရာတွင် ၎င်းတို့သည် အဆင်ပြေစွာ initialize လုပ်ထားသော pointer များဖြစ်ပြီး t ကို အဆုံးသတ်သော '\0' ကို s သို့ ကူးယူပြီးသည်အထိ array များတစ်လျှောက် တစ်ကြိမ်လျှင် character တစ်ခုစီ ချီတက်သွားကြသည်။

လက်တွေ့တွင် strcpy ကို ကျွန်ုပ်တို့ အထက်တွင် ပြသခဲ့သည့်အတိုင်း မရေးကြပါ။ အတွေ့အကြုံရှိသော C programmer များသည် ဤသို့ကို ပို၍နှစ်သက်ကြမည်

```
/* strcpy: t ကို s သို့ ကူးယူသည်; pointer ဗားရှင်း ၂ */

void strcpy(char *s, char *t)

{
	while ((*s++ = *t++) != '\0');
}
```

ဤသည်မှာ s နှင့် t ၏ increment ကို loop ၏ test အစိတ်အပိုင်းသို့ ရွှေ့သည်။ *t++ ၏ တန်ဖိုးသည် t ကို တိုးမီ t point လုပ်နေသော character ဖြစ်သည်။ postfix ++ သည် ဤ character ကို ရယူပြီးမှသာ t ကို ပြောင်းလဲသည်။ တူညီသောနည်းဖြင့် character ကို s တိုးမီ old s position သို့ သိမ်းဆည်းသည်။ ဤ character သည်လည်း loop ကို ထိန်းချုပ်ရန် '\0' နှင့် နှိုင်းယှဉ်သော တန်ဖိုးဖြစ်သည်။ အသားတင်အကျိုးသက်ရောက်မှုမှာ character များကို t မှ s သို့ အဆုံးသတ် '\0' အထိနှင့် အပါအဝင် ကူးယူခြင်းဖြစ်သည်။

နောက်ဆုံး အတိုကောက်အနေဖြင့် '\0' နှင့် နှိုင်းယှဉ်ခြင်းသည် ထပ်နေကြောင်း သတိပြုပါ၊ အကြောင်းမှာ မေးခွန်းမှာ expression သည် သုည (zero) ဖြစ်မဖြစ်သာ ဖြစ်သောကြောင့်ဖြစ်သည်။ ထို့ကြောင့် function ကို

```
/* strcpy: t ကို s သို့ ကူးယူသည်; pointer ဗားရှင်း ၃ */

void strcpy(char *s, char *t)
{
	while (*s++ = *t++);
}
```

ဟု ရေးနိုင်သည်။

ပထမတွေ့ရှိချိန်တွင် ဤသည်မှာ လျှို့ဝှက်ဆန်းကြယ်ပုံရသော်လည်း notation အဆင်ပြေမှုသည် များစွာရှိပြီး idiom ကို ကျွမ်းကျင်သင့်သည်၊ အကြောင်းမှာ C program များတွင် မကြာခဏ မြင်တွေ့ရမည့်အတွက်ဖြစ်သည်။

Standard library (<string.h>) ရှိ strcpy သည် ၎င်း၏ function value အဖြစ် target string ကို return ပြန်သည်။

ကျွန်ုပ်တို့ စမ်းစစ်မည့် ဒုတိယ routine မှာ strcmp(s,t) ဖြစ်ပြီး character string s နှင့် t တို့ကို နှိုင်းယှဉ်ကာ s သည် lexicographically t ထက် နည်းလျှင် အနုတ်၊ တူညီလျှင် သုည (zero) သို့မဟုတ် ကြီးလျှင် အပေါင်းကို return ပြန်သည်။ တန်ဖိုးကို s နှင့် t မတူညီသော ပထမ position ရှိ character များကို နုတ်ခြင်းဖြင့် ရရှိသည်။

```
/* strcmp: s<t ဆိုလျှင် <0၊ s==t ဆိုလျှင် 0၊ s>t ဆိုလျှင် >0 ကို return ပြန်သည် */

int strcmp(char *s, char *t)

{
int i;
for (i = 0; s[i] == t[i]; i++)
	if (s[i] == '\0')
	return 0;
return s[i] - t[i];
}
```

strcmp ၏ pointer ဗားရှင်း:

```
/* strcmp: s<t ဆိုလျှင် <0၊ s==t ဆိုလျှင် 0၊ s>t ဆိုလျှင် >0 ကို return ပြန်သည် */

int strcmp(char *s, char *t)

{
for ( ; *s == *t; s++, t++)

if (*s == '\0')
	return 0;
	return *s - *t;
}
```

++ နှင့် -- တို့သည် prefix သို့မဟုတ် postfix operator များဖြစ်သောကြောင့် * နှင့် ++ နှင့် -- တို့၏ အခြားပေါင်းစပ်မှုများလည်း ဖြစ်ပေါ်သည်၊ သို့သော် နည်းနည်းပါးပါးသာဖြစ်သည်။ ဥပမာအားဖြင့်

```
*--p
```

သည် p point လုပ်သော character ကို မရယူမီ p ကို နုတ်သည်။ တကယ်တမ်း expression အတွဲ

```
*p++ = val; /* val ကို stack ပေါ်သို့ push လုပ်သည် */

val = *--p; /* stack ထိပ်ကို val သို့ pop လုပ်သည် */
```

တို့သည် stack ကို pushing နှင့် popping အတွက် standard idiom ဖြစ်သည်။ Section 4.3 ကိုကြည့်ပါ။

<string.h> header တွင် ဤ section တွင် ဖော်ပြထားသော function များအတွက် declaration များ ပါဝင်သည်၊ ထို့အပြင် standard library မှ အခြား string-handling function အမျိုးမျိုး ပါဝင်သည်။

**လေ့ကျင့်ခန်း ၅-၃။** Chapter 2 တွင် ကျွန်ုပ်တို့ ပြသခဲ့သော function strcat ၏ pointer ဗားရှင်းကို ရေးပါ- strcat(s,t) သည် string t ကို s ၏ အဆုံးသို့ ကူးယူသည်။

**လေ့ကျင့်ခန်း ၅-၄။** Function strend(s,t) ကို ရေးပါ၊ ၎င်းသည် string t သည် string s ၏ အဆုံးတွင် ပေါ်လာလျှင် 1 ကို return ပြန်ပြီး သို့မဟုတ်လျှင် သုည (zero) ကို return ပြန်သည်။

**လေ့ကျင့်ခန်း ၅-၅။** Library function များ strncpy, strncat နှင့် strncmp တို့၏ ဗားရှင်းများကို ရေးပါ၊ ၎င်းတို့သည် ၎င်းတို့၏ argument string များ၏ ပထမ n character များအထိသာ လုပ်ဆောင်သည်။ ဥပမာအားဖြင့် strncpy(s,t,n) သည် t ၏ n character များအထိ s သို့ ကူးယူသည်။ အပြည့်အစုံ ဖော်ပြချက်များသည် Appendix B တွင် ရှိသည်။

**လေ့ကျင့်ခန်း ၅-၆။** အစောပိုင်း chapter များနှင့် လေ့ကျင့်ခန်းများမှ သင့်လျော်သော program များကို array indexing အစား pointer များဖြင့် ပြန်ရေးပါ။ ကောင်းမွန်သော ဖြစ်နိုင်ခြေများတွင် getline (Chapter 1 နှင့် 4)၊ atoi, itoa နှင့် ၎င်းတို့၏ variant များ (Chapter 2, 3 နှင့် 4)၊ reverse (Chapter 3) နှင့် strindex နှင့် getop (Chapter 4) တို့ ပါဝင်သည်။

## ၅.၆ Pointer Array များ; Pointer များသို့ Pointer များ

Pointer များသည် ကိုယ်တိုင်ပင် variable များဖြစ်သောကြောင့် အခြား variable များကဲ့သို့ပင် ၎င်းတို့ကို array များတွင် သိမ်းဆည်းနိုင်သည်။ Text line များ၏ အစုတစ်ခုကို အက္ခရာစဉ် အစီအစဉ်အတိုင်း စီစဉ်မည့် program တစ်ခု ရေးခြင်းဖြင့် ဖော်ပြကြပါစို့၊ UNIX program sort ၏ လျှော့ပြီး ရိုးရှင်းသော ဗားရှင်းဖြစ်သည်။

Chapter 3 တွင် ကျွန်ုပ်တို့သည် integer များ၏ array တစ်ခုကို စီစဉ်မည့် Shell sort function တစ်ခုကို တင်ပြခဲ့ပြီး Chapter 4 တွင် quicksort ဖြင့် ၎င်းကို တိုးတက်စေခဲ့သည်။ တူညီသော algorithm များသည် အလုပ်လုပ်မည်ဖြစ်သော်လည်း ယခုအခါ ကျွန်ုပ်တို့သည် မတူညီသော အရှည်များရှိပြီး integer များနှင့်မတူဘဲ တစ်ခု operation တည်းဖြင့် နှိုင်းယှဉ် သို့မဟုတ် ရွှေ့၍မရသော text line များကို ကိုင်တွယ်ရမည်။ ကျွန်ုပ်တို့သည် variable-length text line များနှင့် ထိရောက်ပြီး အဆင်ပြေစွာ ကိုင်တွယ်မည့် data representation တစ်ခု လိုအပ်သည်။

ဤနေရာတွင် pointer များ၏ array ပါဝင်လာသည်။ စီစဉ်ရမည့် line များကို character array ရှည်တစ်ခုတွင် အဆုံးချင်းတွဲ၍ သိမ်းဆည်းထားလျှင် line တစ်ခုချင်းစီကို ၎င်း၏ ပထမ character သို့ pointer တစ်ခုဖြင့် ရယူအသုံးပြုနိုင်သည်။ Pointer များကိုယ်တိုင်ကို array တစ်ခုတွင် သိမ်းဆည်းနိုင်သည်။ Line နှစ်ခုကို ၎င်းတို့၏ pointer များကို strcmp သို့ ပို့ခြင်းဖြင့် နှိုင်းယှဉ်နိုင်သည်။ အစီအစဉ်မမှန်သော line နှစ်ခုကို လဲလှယ်ရသောအခါ pointer array ရှိ pointer များကို လဲလှယ်ရသည်၊ text line များ ကိုယ်တိုင်ကို မဟုတ်ပါ။

ဤသည်မှာ line များကိုယ်တိုင်ကို ရွှေ့ခြင်းနှင့်အတူ လိုက်မည့် ရှုပ်ထွေးသော storage management နှင့် မြင့်မားသော overhead တို့၏ twin problem များကို ဖယ်ရှားပေးသည်။

Sorting လုပ်ငန်းစဉ်တွင် အဆင့်သုံးဆင့်ရှိသည်:

```
input ၏ line အားလုံးကို ဖတ်သည်

၎င်းတို့ကို စီစဉ်သည်

၎င်းတို့ကို အစီအစဉ်အတိုင်း ပုံနှိပ်သည်
```

ပုံမှန်အတိုင်း program ကို ဤသဘာဝ ခွဲခြားမှုနှင့် ကိုက်ညီသော function များအဖြစ် ခွဲရန် အကောင်းဆုံးဖြစ်ပြီး main routine သည် အခြား function များကို ထိန်းချုပ်သည်။ Sorting အဆင့်ကို ခဏ ရွှေ့ဆိုင်းထားပြီး data structure နှင့် input နှင့် output အပေါ် အာရုံစိုက်ကြပါစို့။

Input routine သည် line တစ်ခုချင်းစီ၏ character များကို စုဆောင်းပြီး သိမ်းဆည်းရမည်၊ ပြီးနောက် line များသို့ pointer များ၏ array တစ်ခုကို တည်ဆောက်ရမည်။ ၎င်းသည် input line များ၏ အရေအတွက်ကိုလည်း ရေတွက်ရမည်၊ အကြောင်းမှာ ထိုအချက်အလက်သည် sorting နှင့် printing အတွက် လိုအပ်သောကြောင့်ဖြစ်သည်။ Input function သည် ကန့်သတ်ထားသော အရေအတွက်သော input line များကိုသာ ကိုင်တွယ်နိုင်သောကြောင့် input အလွန်များလျှင် -1 ကဲ့သို့ တရားမဝင်သော count တစ်ခုကို return ပြန်နိုင်သည်။

Output routine သည် pointer များ၏ array တွင် ပေါ်သော အစီအစဉ်အတိုင်း line များကို ပုံနှိပ်ရုံသာဖြစ်သည်။

```
#include
#include
#define MAXLINES 5000 /* စီစဉ်ရမည့် #line များ၏ အများဆုံး */

char *lineptr[MAXLINES]; /* text line များသို့ pointer များ */
int readlines(char *lineptr[], int nlines);
void writelines(char *lineptr[], int nlines);
void qsort(char *lineptr[], int left, int right);

/* input line များကို စီစဉ်သည် */

main()

{
	int nlines; /* ဖတ်ထားသော input line များ၏ အရေအတွက် */
	if ((nlines = readlines(lineptr, MAXLINES)) >= 0) {
	qsort(lineptr, 0, nlines-1);
	writelines(lineptr, nlines);

return 0;

} else {
	printf("error: input too big to sort\n");
	return 1;
}
}

#define MAXLEN 1000 /* မည်သည့် input line ၏ အများဆုံး အရှည် */

int getline(char *, int);
char *alloc(int);

int readlines(char *lineptr[], int maxlines)

{
	int len, nlines;
	char *p, line[MAXLEN];
	nlines = 0;

while ((len = getline(line, MAXLEN)) > 0)

	if (nlines >= maxlines || p = alloc(len) == NULL)
	return -1;

else {
	line[len-1] = '\0'; /* newline ကို ဖျက်သည် */ strcpy(p, line);
	lineptr[nlines++] = p;
}
	return nlines;
}

	int i;
	for (i = 0; i < nlines; i++)

}
```

ဟု ရေးနိုင်သည်။

အစတွင် *lineptr သည် ပထမ line ကို point လုပ်သည်။ element တစ်ခုလျှင် nlines ကို ရေတွက်ကျဆင်းနေစဉ် ၎င်းကို နောက်ထပ် line pointer သို့ တိုးသည်။

Input နှင့် output ကို ထိန်းချုပ်ထားပြီးသောကြောင့် ကျွန်ုပ်တို့သည် sorting သို့ ဆက်လုပ်နိုင်သည်။ Chapter 4 မှ quicksort သည် အနည်းငယ် ပြောင်းလဲမှုများ လိုအပ်သည်- declaration များကို ပြင်ဆင်ရမည်၊ ပြီးနောက် comparison operation ကို strcmp ကို ခေါ်ဆိုခြင်းဖြင့် လုပ်ရမည်။

Algorithm သည် တူညီနေသောကြောင့် ၎င်းသည် ဆက်အလုပ်လုပ်မည်ဟု ကျွန်ုပ်တို့ အနည်းငယ် ယုံကြည်မှုရှိစေသည်။

```
/* qsort: v[left]...v[right] ကို တိုးလာသော အစီအစဉ်အဖြစ် စီစဉ်သည် */

void qsort(char *v[], int left, int right)

{
int i, last;

void swap(char *v[], int i, int j);

if (left >= right) /* array တွင် နှစ်ခုထက်နည်းသော */

	return; /* element များရှိလျှင် ဘာမှမလုပ်ပါ */
	swap(v, left, (left + right)/2);
	last = left;

for (i = left+1; i <= right; i++)

if (strcmp(v[i], v[left]) < 0)
	swap(v, ++last, i);
	swap(v, left, last);
	qsort(v, left, last-1);
	qsort(v, last+1, right);
}
```

တူညီစွာ swap routine သည် အသေးအဖွဲ ပြောင်းလဲမှုများသာ လိုအပ်သည်:

```
/* swap: v[i] နှင့် v[j] ကို လဲလှယ်သည် */

void swap(char *v[], int i, int j)
{
	char *temp;
	temp = v[i];
	v[i] = v[j];
	v[j] = temp;
}
```

v (lineptr ၏ အမည်ပြောင်း) ၏ တစ်ခုချင်းစီသော element သည် character pointer ဖြစ်သောကြောင့် temp သည်လည်း ဖြစ်ရမည်၊ ထို့ကြောင့် တစ်ခုကို နောက်တစ်ခုသို့ ကူးယူနိုင်သည်။
